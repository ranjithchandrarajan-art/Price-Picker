// ============================================================ // AUCTION VEHICLE PRICE EVALUATOR — Google Apps Script REST API // ============================================================ // SETUP: Replace with your actual Google Sheet ID const SHEET_ID = "YOUR_GOOGLE_SHEET_ID_HERE"; const DATA_SHEET = "DATA"; const USERS_SHEET = "USERS"; // ── Column indices (1-based) for DATA sheet ────────────────── const COL = { INDEX: 1, EVALUATION_DATE: 2, REGISTRATION_NUMBER: 3, STAGE: 4, CUSTOMER_NAME: 5, EVALUATOR_NAME: 6, MAKE_MODEL: 7, COLOR_ID: 8, FUEL_TYPE: 9, TRANSMISSION: 10, YEAR_OF_MFG: 11, INSURANCE_END_DATE: 12, RC_REGN_DATE: 13, VEHICLE_OWNER_NUMBER: 14, SPEEDOMETER: 15, TIMESTAMP: 16, STATUS: 17, TVSC_PRICE: 18, EXPECTED_PRICE: 19, START_PRICE: 20 }; const USER_COL = { MOBILE: 1, NAME: 2, ROLE: 3 }; // ── JSON response helper ────────────────────────────────────── function jsonOut(data) { return ContentService .createTextOutput(JSON.stringify(data)) .setMimeType(ContentService.MimeType.JSON); } // ============================================================ // GET → ?action=login&mobile=XXXXXXXXXX // → ?action=vehicles&name=NAME&role=ROLE // → ?action=pivot // ============================================================ function doGet(e) { try { const p = e.parameter; if (p.action === "login") return jsonOut(loginByMobile(p.mobile)); if (p.action === "vehicles") return jsonOut(getVehicles(p.name, p.role)); if (p.action === "pivot") return jsonOut(getAdminPivot()); return jsonOut({ success: false, message: "Unknown action." }); } catch (err) { return jsonOut({ success: false, message: err.message }); } } // ============================================================ // POST → { action:"save", rowIndex, tvscPrice, expectedPrice } // ============================================================ function doPost(e) { try { const b = JSON.parse(e.postData.contents); if (b.action === "save") return jsonOut(savePrices(b.rowIndex, b.tvscPrice, b.expectedPrice)); return jsonOut({ success: false, message: "Unknown action." }); } catch (err) { return jsonOut({ success: false, message: err.message }); } } // ── AUTH ────────────────────────────────────────────────────── function loginByMobile(mobile) { if (!mobile) return { success: false, message: "Mobile is required." }; const data = SpreadsheetApp.openById(SHEET_ID).getSheetByName(USERS_SHEET).getDataRange().getValues(); for (let i = 1; i < data.length; i++) { if (String(data[i][USER_COL.MOBILE - 1]).trim() === String(mobile).trim()) { return { success: true, user: { mobile: String(data[i][USER_COL.MOBILE - 1]).trim(), name: String(data[i][USER_COL.NAME - 1]), role: String(data[i][USER_COL.ROLE - 1]) }}; } } return { success: false, message: "Mobile number not registered." }; } // ── FETCH VEHICLES ──────────────────────────────────────────── function getVehicles(userName, role) { const data = SpreadsheetApp.openById(SHEET_ID).getSheetByName(DATA_SHEET).getDataRange().getValues(); const rows = []; for (let i = 1; i < data.length; i++) { const r = data[i]; if (!r[COL.REGISTRATION_NUMBER - 1]) continue; if (role !== "Admin" && String(r[COL.EVALUATOR_NAME - 1]).trim() !== String(userName).trim()) continue; rows.push({ rowIndex: i + 1, index: r[COL.INDEX - 1], evaluationDate: fmt(r[COL.EVALUATION_DATE - 1]), registrationNumber: r[COL.REGISTRATION_NUMBER - 1], stage: r[COL.STAGE - 1], customerName: r[COL.CUSTOMER_NAME - 1], evaluatorName: r[COL.EVALUATOR_NAME - 1], makeModel: r[COL.MAKE_MODEL - 1], colorId: r[COL.COLOR_ID - 1], fuelType: r[COL.FUEL_TYPE - 1], transmission: r[COL.TRANSMISSION - 1], yearOfMfg: r[COL.YEAR_OF_MFG - 1], insuranceEndDate: fmt(r[COL.INSURANCE_END_DATE - 1]), rcRegnDate: fmt(r[COL.RC_REGN_DATE - 1]), vehicleOwnerNumber: r[COL.VEHICLE_OWNER_NUMBER - 1], speedometer: r[COL.SPEEDOMETER - 1], timestamp: fmt(r[COL.TIMESTAMP - 1]), status: r[COL.STATUS - 1] || "Pending", tvscPrice: r[COL.TVSC_PRICE - 1] || "", expectedPrice: r[COL.EXPECTED_PRICE - 1] || "", startPrice: r[COL.START_PRICE - 1] || "" }); } return { success: true, data: rows }; } // ── SAVE ────────────────────────────────────────────────────── function savePrices(rowIndex, tvscPrice, expectedPrice) { const sheet = SpreadsheetApp.openById(SHEET_ID).getSheetByName(DATA_SHEET); const sp = calcStartPrice(Number(expectedPrice), Number(tvscPrice)); const now = new Date(); sheet.getRange(rowIndex, COL.TVSC_PRICE).setValue(Number(tvscPrice)); sheet.getRange(rowIndex, COL.EXPECTED_PRICE).setValue(Number(expectedPrice)); sheet.getRange(rowIndex, COL.START_PRICE).setValue(sp); sheet.getRange(rowIndex, COL.STATUS).setValue("Updated"); sheet.getRange(rowIndex, COL.TIMESTAMP).setValue(now); return { success: true, startPrice: sp, timestamp: fmt(now) }; } // ── PIVOT ───────────────────────────────────────────────────── function getAdminPivot() { const data = SpreadsheetApp.openById(SHEET_ID).getSheetByName(DATA_SHEET).getDataRange().getValues(); const pivot = {}; for (let i = 1; i < data.length; i++) { const r = data[i]; if (!r[COL.REGISTRATION_NUMBER - 1]) continue; const ev = r[COL.EVALUATOR_NAME - 1] || "Unassigned"; const st = r[COL.STATUS - 1] || "Pending"; if (!pivot[ev]) pivot[ev] = { updated: 0, pending: 0, total: 0 }; pivot[ev].total++; String(st).toLowerCase() === "updated" ? pivot[ev].updated++ : pivot[ev].pending++; } return { success: true, pivot: pivot }; } // ── HELPERS ─────────────────────────────────────────────────── // Start Price = TVSC Price − (bracket × 2) // bracket: Expected ≥ 7L → 7000 | ≥ 5L → 5000 | else → 2000 function calcStartPrice(expectedPrice, tvscPrice) { if (!expectedPrice || isNaN(expectedPrice)) return ""; if (!tvscPrice || isNaN(tvscPrice)) return ""; const bracket = expectedPrice >= 700000 ? 7000 : expectedPrice >= 500000 ? 5000 : 2000; const multiplier = expectedPrice < 500000 ? 3 : 2; return tvscPrice - (bracket * multiplier); } function fmt(v) { if (!v) return ""; if (v instanceof Date) return Utilities.formatDate(v, Session.getScriptTimeZone(), "dd-MM-yyyy HH:mm"); return String(v); }
